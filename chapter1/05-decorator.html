05-decorator

这种里面包含函数,然后修改他使用前和使用后的结果就叫装饰器

  • 任务添加和删除的时间
from datetime import datetime

def display_time():
    print(str(datetime.now()))

class Schedule(object):

    name = None
    desp = None
    tasks = None

    def __init__(self, n, d):
        self.name = n
        self.desp = d
        self.tasks = {}

    def add_task(self, name, content, priority):
        if name not in self.tasks:
            task = [name, priority, content]
            self.tasks.update({name: task})
        self.display()
        display_time()

    def remove_task(self, name):
        if name in self.tasks:
            self.tasks.pop(name, None)
        self.display()
        display_time()

    def display(self):
        if self.tasks:
            print('here are your tasks:')
            for v in self.tasks.values():
                print('%-10s %-5s %-50s' % (v[0], v[2], v[1]))

    def __str__(self):
        return 'Schedule %s, %s' % (self.name, self.desp)

s = Schedule('s1', 'ITschool')
s.add_task('cs', 'teach introduction to python course', 1000)

python中函数是一级对象,可以将一个函数作为参数传递到另一个函数中

  • 将函数作为参数传进另一个函数
# coding=UTF-8

# 函数里面定义了一个函数 
# 因为函数是对象 所以函数里面可以定义函数;好比对象里面定义函数 
def answer_to_university():
    def give_42():
        return 42
    # 函数answer_to_university的返回值是一个函数
    return give_42

# 将函数赋值给变量,并且调用他
answer = answer_to_university()

print(answer_to_university) # atu function
print(answer_to_university()) # give_42 function 
print(answer_to_university()()) # 42
print(answer()) # 42
  • 函数和字符串 数字是一样的使用 只不过函数后面多一个()

# coding=UTF-8

def display_decorator(func):
    def wrapper():
        print('before func')
        func()
        print('after func')
    return wrapper

def test_function():
    print("this is awesome function")

# 直接使用函数test_function
test_function()

# 通过装饰器使用函数
display_decorator(test_function)() # b t a 

# display_decorator就是装饰器
# 他里面包含一个函数 
# 并且修改该函数使用之前和之后(添加了before和after) 
# 这就是装饰器

这种里面包含函数,然后修改他使用前和使用后的结果就叫装饰器

  • python为装饰器的易用性提供的语法如下
# coding=UTF-8

# 这个函数是装饰器
def display_decorator(func):
    def wrapper():
        print('before func')
        func()
        print('after func')
    return wrapper

# 这里是使用装饰器
@display_decorator
def test_function():
    print("this is awesome function")

# 使用装饰器后函数的调用
test_function() # b t a

# coding=UTF-8

# 装饰器 
# 添加日期
from datetime import datetime

# display-decorator就是装饰器
def display_decorator(func):
    # *非关键字收集参数,**关键字收集参数 
    def wrapper(*args, **kwargs):
        print(str(datetime.now()))
        return func(*args, **kwargs)
    return wrapper


class Schedule(object):

    name = None
    desp = None
    tasks = None

    def __init__(self, n, d):
        self.name = n
        self.desp = d
        self.tasks = {}

# @使用装饰器的快捷键
    @display_decorator
    def add_task(self, name, content, priority):
        if name not in self.tasks:
            task = [name, priority, content]
            self.tasks.update({name:task})
        self.display()

    @display_decorator
    def remove_task(self, name):
        if name in self.tasks:
            self.tasks.pop(name,None)
        self.display()

    def display(self):
        if self.tasks:
            print("here are your tasks:")
            for v in self.tasks.values():
                print("%-10s %-5s %-50s" % (v[0], v[2], v[1]))

    def  __str__(self):
        return "Schedule %s , %s" % (self.name,self.desp)


s = Schedule('s1','ITshcool')
s.add_task('cs','teach introduction to python course',1000)


print(s)

@装饰器 放到函数前面是使用装饰器的快捷键 *和** 是非关键词参数收集和关键词参数收集,用于收集列表和字典
