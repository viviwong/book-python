08-command

python脚本通常使用命令行调用

  • 新建一个文件touch command.py,通过命令行执行python command.py 1 2
# coding=UTF-8
import sys

if __name__ == '__main__':
    # python将文件名 参数1 参数2 ...以列表的形式存放在sys.argv中
    args = sys.argv
    a1 = args[1]
    a2 = args[2]

    print(args,a1,a2) # (['command.py', '1', '2'], '1', '2')

1 通过sys.agv可以获得一个列表
2 python command.py hello world
3 args[0] = command.py args[1] = hello args[2] = world

  • 认识if __name__ == '__main__':
# coding=UTF-8
import sys

print("1 作为模块运行的时候被调用")

# 当command.py直接运行时,if之下的内容被调用执行;当command.py作为模块被调用时,if之下的内容不被执行
if __name__ == '__main__':
    # python将文件名 参数1 参数2 ...以列表的形式存放在sys.argv中
    args = sys.argv
    a1 = args[1]
    a2 = args[2]

    print(args,a1,a2) # (['command.py', '1', '2'], '1', '2')

    print("2 command.py直接运行时候被调用")
  • 命令行执行python command.py 1 2结果为 
1 作为模块运行的时候被调用
(['command.py', '1', '2'], '1', '2')
2 command.py直接运行时候被调用
